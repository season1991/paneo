## 1. 技术选型

| 层级 | 技术 | 版本 | 选型理由 |
|------|------|------|----------|
| 桌面框架 | Electron | 42.4.0 | 唯一成熟支持 `<webview>` 标签的方案；Chromium 内核保证 AI 网站兼容性；跨平台 |

#### InjectService

```typescript
// src/main/injector.ts

/**
 * 主注入入口：优先使用 OS 级输入模拟，失败时降级到 JS 注入
 */
async function injectToPanel(
  webview: Electron.WebviewTag,
  text: string,
  selectors: SiteConfig['selectors']
): Promise<InjectResult> {
  const simResult = await injectViaInputSimulation(webview, text, selectors);
  if (simResult.status === 'ok') return simResult;
  return injectViaJS(webview, text, selectors);
}

/**
 * OS 级输入模拟（首选方案）
 * 通过 sendInputEvent 发送真实的键盘/鼠标事件
 */
async function injectViaInputSimulation(
  webview: Electron.WebviewTag,
  text: string,
  selectors: SiteConfig['selectors']
): Promise<InjectResult> {
  try {
    const inputPos = await webview.executeJavaScript(`
      (function() {
        const sels = ${JSON.stringify(selectors.input)};
        for (const sel of sels) {
          const el = document.querySelector(sel);
          if (el) {
            el.scrollIntoView({ block: 'center' });
            const r = el.getBoundingClientRect();
            return { x: Math.round(r.left + r.width / 2), y: Math.round(r.top + r.height / 2) };
          }
        }
        return null;
      })()
    `);
    if (!inputPos) return { id: '', status: 'fail', reason: 'INPUT_NOT_FOUND' };

    // 点击聚焦
    webview.sendInputEvent({ type: 'mouseDown', x: inputPos.x, y: inputPos.y, button: 'left', clickCount: 1 });
    webview.sendInputEvent({ type: 'mouseUp', x: inputPos.x, y: inputPos.y, button: 'left', clickCount: 1 });
    await sleep(100);

    // 逐字符输入
    for (const char of text) {
      webview.sendInputEvent({ type: 'keyDown', keyCode: char });
      webview.sendInputEvent({ type: 'char', keyCode: char });
      webview.sendInputEvent({ type: 'keyUp', keyCode: char });
      await sleep(30 + Math.random() * 50);
    }
    await sleep(200);

    // 触发发送
    if (selectors.sendButton?.length) {
      const btnPos = await webview.executeJavaScript(`
        (function() {
          const sels = ${JSON.stringify(selectors.sendButton)};
          for (const sel of sels) {
            const btn = document.querySelector(sel);
            if (btn && !btn.disabled) {
              btn.scrollIntoView({ block: 'center' });
              const r = btn.getBoundingClientRect();
              return { x: Math.round(r.left + r.width / 2), y: Math.round(r.top + r.height / 2) };
            }
          }
          return null;
        })()
      `);
      if (btnPos) {
        webview.sendInputEvent({ type: 'mouseDown', x: btnPos.x, y: btnPos.y, button: 'left', clickCount: 1 });
        webview.sendInputEvent({ type: 'mouseUp', x: btnPos.x, y: btnPos.y, button: 'left', clickCount: 1 });
      } else {
        simulateEnter(webview);
      }
    } else {
      simulateEnter(webview);
    }
    return { id: '', status: 'ok' };
  } catch {
    return { id: '', status: 'fail', reason: 'EXEC_ERROR' };
  }
}

function simulateEnter(webview: Electron.WebviewTag) {
  webview.sendInputEvent({ type: 'keyDown', keyCode: 'Enter' });
  webview.sendInputEvent({ type: 'char', keyCode: '\r' });
  webview.sendInputEvent({ type: 'keyUp', keyCode: 'Enter' });
}

/**
 * JS 注入（降级方案）
 */
function injectViaJS(webview: Electron.WebviewTag, text: string, selectors: SiteConfig['selectors']): Promise<InjectResult> {
  return webview.executeJavaScript(buildInjectScript(text, selectors))
    .then(() => ({ id: '', status: 'ok' as const }))
    .catch(() => ({ id: '', status: 'fail' as const, reason: 'EXEC_ERROR' }));
}

function buildInjectScript(text: string, selectors: SiteConfig['selectors']): string {
  return `
    (function() {
      const text = ${JSON.stringify(text)};
      const inputSelectors = ${JSON.stringify(selectors.input)};
      const sendButtonSelectors = ${JSON.stringify(selectors.sendButton || [])};
      function findInput(sels) {
        for (const sel of sels) {
          const el = document.querySelector(sel);
          if (!el) continue;
          if (el.tagName === 'TEXTAREA' || el.tagName === 'INPUT') return { el, type: 'textarea' };
          if (el.isContentEditable) return { el, type: 'contenteditable' };
        }
        return null;
      }
      function fillInput(found, text) {
        if (found.type === 'textarea') {
          Object.getOwnPropertyDescriptor(window.HTMLTextAreaElement.prototype, 'value').set.call(found.el, text);
          found.el.dispatchEvent(new Event('input', { bubbles: true }));
        } else {
          found.el.focus();
          found.el.innerHTML = text;
          found.el.dispatchEvent(new InputEvent('input', { bubbles: true, inputType: 'insertText' }));
        }
      }
      function triggerSend(found, sels) {
        for (const sel of sels) {
          const btn = document.querySelector(sel);
          if (btn && !btn.disabled) { btn.click(); return; }
        }
        found.el.dispatchEvent(new KeyboardEvent('keydown', { key: 'Enter', code: 'Enter', keyCode: 13, bubbles: true }));
      }
      const found = findInput(inputSelectors);
      if (!found) throw new Error('INPUT_NOT_FOUND');
      fillInput(found, text);
      triggerSend(found, sendButtonSelectors);
      return { status: 'ok' };
    })();
  `;
}
```

#### CookieService

```typescript
// src/main/cookie.ts
import { session } from 'electron';

class CookieService {
  /**
   * 从指定 webview 的 session 中提取所有 cookie，序列化为字符串
   * @param partition webview 的 partition 标识，如 "persist:chatgpt"
   * @returns 序列化的 cookie 字符串（JSON 格式）
   */
  async extractCookies(partition: string): Promise<string> {
    const sess = session.fromPartition(partition);
    const cookies = await sess.cookies.get({});
    return JSON.stringify(cookies.map(c => ({
      name: c.name,
      value: c.value,
      domain: c.domain,
      path: c.path,
      secure: c.secure,
      httpOnly: c.httpOnly,
      expirationDate: c.expirationDate,
    })));
  }

  /**
   * 将已保存的 cookie 注入到指定 webview 的 session 中
   * 在 webview 加载 URL 之前调用，确保网站能识别登录态
   * @param partition webview 的 partition 标识
   * @param cookieJson 序列化的 cookie 字符串
   * @param url 目标网站 URL（用于设置 cookie 的 domain）
   */
  async injectCookies(partition: string, cookieJson: string, url: string): Promise<void> {
    const sess = session.fromPartition(partition);
    const cookies = JSON.parse(cookieJson);
    const urlObj = new URL(url);

    for (const cookie of cookies) {
      try {
        await sess.cookies.set({
          url: `${urlObj.protocol}//${cookie.domain || urlObj.hostname}${cookie.path || '/'}`,
          name: cookie.name,
          value: cookie.value,
          domain: cookie.domain,
          path: cookie.path || '/',
          secure: cookie.secure,
          httpOnly: cookie.httpOnly,
          expirationDate: cookie.expirationDate,
        });
      } catch (e) {
        // 某些 cookie 可能因 domain 不匹配而设置失败，忽略
        console.warn(`Failed to inject cookie ${cookie.name}:`, e);
      }
    }
  }

  /**
   * 清除指定 webview session 的所有 cookie
   * @param partition webview 的 partition 标识
   */
  async clearCookies(partition: string): Promise<void> {
    const sess = session.fromPartition(partition);
    await sess.cookies.flushStore();
    const cookies = await sess.cookies.get({});
    for (const cookie of cookies) {
      const protocol = cookie.secure ? 'https:' : 'http:';
      await sess.cookies.remove(`${protocol}//${cookie.domain}${cookie.path}`, cookie.name);
    }
  }
}

export const cookieService = new CookieService();
```

```typescript
import { contextBridge, ipcRenderer } from 'electron';

contextBridge.exposeInMainWorld('paneo', {

  // Cookie 操作
  extractCookies: (partition: string): Promise<string> =>
    ipcRenderer.invoke('extract-cookies', partition),
  injectCookies: (partition: string, cookieJson: string, url: string): Promise<void> =>
    ipcRenderer.invoke('inject-cookies', partition, cookieJson, url),
  clearCookies: (partition: string): Promise<void> =>
    ipcRenderer.invoke('clear-cookies', partition),

  // 自动分析
  analyzeSite: (partition: string): Promise<Selectors> =>
    ipcRenderer.invoke('analyze-site', partition),
  verifySelectors: (partition: string, selectors: Selectors): Promise<boolean> =>
    ipcRenderer.invoke('verify-selectors', partition, selectors),

  onInjectResult: (callback: (result: InjectResult) => void) => {
    ipcRenderer.on('inject-result', (_, result) => callback(result));

    ipcRenderer.removeAllListeners('inject-result');
  },
```

```tsx
// src/renderer/components/WebViewPanel.tsx
function WebViewPanel({ site, status, onClose }: WebViewPanelProps) {
  const webviewRef = useRef<Electron.WebviewTag>(null);
  const [loadState, setLoadState] = useState<'loading' | 'ready' | 'error'>('loading');

  // 启动时注入已保存的 cookie，然后再加载 URL
  useEffect(() => {
    if (site.cookie && webviewRef.current) {
      const partition = `persist:${site.id}`;
      window.paneo.injectCookies(partition, site.cookie, site.url).then(() => {
        webviewRef.current?.loadURL(site.url);
      });
    }
  }, [site.id]);

  return (
    <div className={`panel panel--${loadState}`}>
      {/* 面板头部 */}
      <div className="panel__header">
        <span className="panel__name">{site.name}</span>
        <div className="panel__nav">
          <button onClick={() => webviewRef.current?.goBack()} title="后退">←</button>
          <button onClick={() => webviewRef.current?.goForward()} title="前进">→</button>
          <button onClick={() => webviewRef.current?.reload()} title="刷新">↻</button>
        </div>
        <button className="panel__close" onClick={onClose} title="移除">×</button>
        {/* 注入状态指示器 */}
        {status === 'inject-fail' && <span className="panel__dot panel__dot--fail" title="自动输入不可用">●</span>}
        {status === 'skipped' && <span className="panel__dot panel__dot--skip" title="面板未就绪">○</span>}
      </div>

      {/* 注入失败横幅 */}
      {status === 'inject-fail' && (
        <div className="panel__banner">
          ⚠ 未能自动输入，请手动在此输入框中粘贴你的问题。
          <button onClick={() => {}}>关闭</button>
        </div>
      )}

      {/* 加载失败 */}
      {loadState === 'error' && (
        <div className="panel__error">
          <p>✕ 加载失败</p>
          <p>可能原因：网络中断 / 网站不可用</p>
          <button onClick={() => webviewRef.current?.reload()}>重新加载</button>
        </div>
      )}

      {/* 加载中 */}
      {loadState === 'loading' && (
        <div className="panel__loading">◌ 加载中...</div>
      )}

      {/* webview：有 cookie 时不自动加载，由 useEffect 控制加载时机 */}
      <webview
        ref={webviewRef}
        src={site.cookie ? 'about:blank' : site.url}
        partition={`persist:${site.id}`}
        allowpopups
        className="panel__webview"
        style={{ display: loadState === 'ready' ? 'flex' : 'none' }}
        onDidFinishLoad={() => setLoadState('ready')}
        onDidFailLoad={() => setLoadState('error')}
      />
    </div>
  );
}
```

```tsx
// src/renderer/components/InputBar.tsx
function InputBar({ onSend }: { onSend: (text: string) => void }) {
  const [text, setText] = useState('');

  const handleKeyDown = (e: React.KeyboardEvent) => {
    if (e.key === 'Enter' && !e.shiftKey) {
      e.preventDefault();
      if (text.trim()) {
        onSend(text.trim());
        setText('');
      }
    }
  };

  return (
    <div className="input-bar">
      <textarea
        className="input-bar__textarea"
        value={text}
        onChange={e => setText(e.target.value)}
        onKeyDown={handleKeyDown}
        placeholder="输入问题..."
        rows={1}
      />
      <button className="input-bar__clear" onClick={() => setText('')}>清空</button>
      <button className="input-bar__send" onClick={() => { if (text.trim()) { onSend(text.trim()); setText(''); } }}>
        发送 ▶
      </button>
    </div>
  );
}
```

## 6. 注入机制详解

这是 Paneo 最核心的技术难点，单独详细说明。

### 6.1 注入策略总览

Paneo 采用**两级注入策略**，优先使用 OS 级键盘鼠标模拟，降级使用 JS 注入：

| 优先级 | 方式 | 原理 | 优点 | 缺点 |
|--------|------|------|------|------|
| **1（首选）** | OS 级输入模拟 | 通过 `webview.sendInputEvent()` 发送真实的键盘/鼠标事件 | 不受 CSP 限制；无法被网站 JS 检测；与真实用户操作完全一致 | 需要先获取元素坐标；输入速度受打字延迟影响 |
| 2（降级） | JS 注入 | 通过 `webview.executeJavaScript()` 直接操作 DOM | 速度快；不依赖坐标 | 可能被 CSP 拦截；React 等框架需要特殊处理；可被网站检测 |

### 6.2 OS 级输入模拟（首选方案）

利用 Electron `<webview>` 标签的 `sendInputEvent()` API，向 webview 发送操作系统级别的键盘和鼠标事件。这些事件与真实用户操作完全一致，网站无法区分。

#### 6.2.1 整体流程

```text
1. 通过 CSS 选择器找到目标元素
2. 执行 JS 获取元素在 webview 中的屏幕坐标 (getBoundingClientRect)
3. 模拟鼠标点击目标位置（聚焦输入框）
4. 逐字符模拟键盘输入（keydown → char → keyup）
5. 找到发送按钮的坐标
6. 模拟鼠标点击发送按钮
```

#### 6.2.2 获取元素坐标

```javascript
// 在 webview 中执行：获取元素的屏幕坐标
function getElementCenter(selector) {
  const el = document.querySelector(selector);
  if (!el) return null;
  const rect = el.getBoundingClientRect();
  return {
    x: Math.round(rect.left + rect.width / 2),
    y: Math.round(rect.top + rect.height / 2),
    width: rect.width,
    height: rect.height,
  };
}
```

#### 6.2.3 模拟鼠标点击

```typescript
// 主进程：向 webview 发送鼠标事件
function simulateClick(webview: Electron.WebviewTag, x: number, y: number) {
  webview.sendInputEvent({ type: 'mouseDown', x, y, button: 'left', clickCount: 1 });
  webview.sendInputEvent({ type: 'mouseUp', x, y, button: 'left', clickCount: 1 });
}
```

#### 6.2.4 模拟键盘输入（逐字符）

```typescript
// 主进程：逐字符发送键盘事件
async function simulateTyping(webview: Electron.WebviewTag, text: string) {
  for (const char of text) {
    webview.sendInputEvent({ type: 'keyDown', keyCode: char });
    webview.sendInputEvent({ type: 'char', keyCode: char });
    webview.sendInputEvent({ type: 'keyUp', keyCode: char });
    await sleep(30 + Math.random() * 50); // 30~80ms 模拟真实打字节奏
  }
}
```

#### 6.2.5 模拟 Enter 键发送

```typescript
function simulateEnter(webview: Electron.WebviewTag) {
  webview.sendInputEvent({ type: 'keyDown', keyCode: 'Enter' });
  webview.sendInputEvent({ type: 'char', keyCode: '\r' });
  webview.sendInputEvent({ type: 'keyUp', keyCode: 'Enter' });
}
```

#### 6.2.6 完整注入脚本（主进程侧）

```typescript
// src/main/injector.ts
async function injectViaInputSimulation(
  webview: Electron.WebviewTag,
  text: string,
  selectors: SiteConfig['selectors']
): Promise<InjectResult> {
  try {
    // 步骤 1: 在 webview 中找到输入框并获取坐标
    const inputPos = await webview.executeJavaScript(`
      (function() {
        const selectors = ${JSON.stringify(selectors.input)};
        for (const sel of selectors) {
          const el = document.querySelector(sel);
          if (el) {
            el.scrollIntoView({ block: 'center' });
            const rect = el.getBoundingClientRect();
            return { x: Math.round(rect.left + rect.width / 2), y: Math.round(rect.top + rect.height / 2) };
          }
        }
        return null;
      })()
    `);
    if (!inputPos) return { id: '', status: 'fail', reason: 'INPUT_NOT_FOUND' };

    // 步骤 2: 点击输入框（聚焦）
    simulateClick(webview, inputPos.x, inputPos.y);
    await sleep(100);

    // 步骤 3: 逐字符输入文本
    await simulateTyping(webview, text);
    await sleep(200);

    // 步骤 4: 触发发送
    if (selectors.sendButton && selectors.sendButton.length > 0) {
      const btnPos = await webview.executeJavaScript(`
        (function() {
          const selectors = ${JSON.stringify(selectors.sendButton)};
          for (const sel of selectors) {
            const btn = document.querySelector(sel);
            if (btn && !btn.disabled) {
              btn.scrollIntoView({ block: 'center' });
              const rect = btn.getBoundingClientRect();
              return { x: Math.round(rect.left + rect.width / 2), y: Math.round(rect.top + rect.height / 2) };
            }
          }
          return null;
        })()
      `);
      if (btnPos) {
        simulateClick(webview, btnPos.x, btnPos.y);
      } else {
        simulateEnter(webview);
      }
    } else {
      simulateEnter(webview);
    }

    return { id: '', status: 'ok' };
  } catch (e) {
    return { id: '', status: 'fail', reason: 'EXEC_ERROR' };
  }
}
```

### 6.3 JS 注入（降级方案）

当 OS 级输入模拟失败时，回退到 JS 注入。

#### 6.3.1 两类输入框

| 类型 | 代表网站 | 元素 | 注入方式 |
|------|----------|------|----------|
| textarea | ChatGPT, DeepSeek | `<textarea>` | native setter + `input` 事件 |
| contenteditable | Claude, Gemini, Kimi | `<div contenteditable="true">` | `focus()` + `innerHTML` + `InputEvent` |

#### 6.3.2 textarea 注入（React SPA 兼容）

```javascript
const nativeSetter = Object.getOwnPropertyDescriptor(
  window.HTMLTextAreaElement.prototype, 'value'
).set;
nativeSetter.call(el, text);
el.dispatchEvent(new Event('input', { bubbles: true }));
```

#### 6.3.3 contenteditable 注入

```javascript
el.focus();
el.innerHTML = text;
el.dispatchEvent(new InputEvent('input', { bubbles: true, inputType: 'insertText' }));
```

#### 6.3.4 触发发送

```javascript
for (const sel of sendButtonSelectors) {
  const btn = document.querySelector(sel);
  if (btn && !btn.disabled) { btn.click(); return; }
}
el.dispatchEvent(new KeyboardEvent('keydown', { key: 'Enter', code: 'Enter', keyCode: 13, bubbles: true }));
```

```text
用户在统一输入栏按回车

      │
      ▼

      │                                          │
      ▼                                          │

      │                                          │

      │                                          │
     是                                          │

      ▼                                          │

  · simulateTyping（逐字符输入）                   │

      │                                          │
  成功？──是──▶ 标记 OK ✓                         │
      │                                          │
     否                                          │
```

### 6.5 选择器失效应对

当所有选择器都未匹配时，注入脚本抛出 `INPUT_NOT_FOUND` 错误。主进程捕获后通知渲染进程，面板显示"注入失败"状态（小红点 + 横幅提示）。

用户可以通过"编辑网站"功能手动更新选择器。预置选择器的更新随软件版本发布。

### 6.6 输入模拟的注意事项

| 事项 | 说明 |
|------|------|
| 打字延迟 | 每个字符间 30~80ms 随机延迟，模拟真实打字节奏。过快可能导致网站前端来不及响应 |
| 焦点管理 | 点击输入框后需等待 100ms 确保焦点就绪，再开始输入 |
| 滚动对齐 | 输入前先 `scrollIntoView` 确保目标元素在可视区域内，否则坐标计算会偏差 |
| 中文输入 | `sendInputEvent` 的 `char` 类型支持 Unicode 字符，可直接发送中文 |
| 特殊键 | Shift+Enter（换行）、Backspace（删除）等通过 `keyDown`/`keyUp` 模拟 |
| 输入法兼容 | 部分输入法可能拦截键盘事件；如遇问题可尝试通过剪贴板粘贴（Ctrl+V） |

## 7. IPC 协议定义

| 通道 | 方向 | 参数 | 返回值 | 说明 |
|------|------|------|--------|------|
| `send-to-all` | 渲染→主 | `text: string` | `InjectResult[]` | 发送问题到所有面板 |
| `get-sites` | 渲染→主 | 无 | `SiteConfig[]` | 获取网站配置列表 |
| `save-sites` | 渲染→主 | `sites: SiteConfig[]` | `void` | 保存网站配置 |
| `get-layout` | 渲染→主 | 无 | `LayoutConfig` | 获取布局配置 |
| `save-layout` | 渲染→主 | `layout: LayoutConfig` | `void` | 保存布局配置 |
| `inject-result` | 主→渲染 | `InjectResult` | — | 注入结果实时推送（单向） |
| `analyze-site` | 渲染→主 | `partition: string` | `Selectors` | 对 webview 中当前页面执行自动分析，返回最佳选择器 |
| `verify-selectors` | 渲染→主 | `partition: string, selectors: Selectors` | `boolean` | 验证选择器能否成功注入 |
| `extract-cookies` | 渲染→主 | `partition: string` | `string` | 从 webview session 提取 cookie，返回 JSON 字符串 |
| `inject-cookies` | 渲染→主 | `partition: string, cookieJson: string, url: string` | `void` | 将已保存的 cookie 注入到 webview session |
| `clear-cookies` | 渲染→主 | `partition: string` | `void` | 清除指定 webview session 的所有 cookie |
